server {
  server_name concluder.org;

  # HTTPS (managed by Certbot)
  listen [::]:443 ssl; # managed by Certbot
  listen 443 ssl; # managed by Certbot
  ssl_certificate /etc/letsencrypt/live/concluder.org/fullchain.pem; # managed by Certbot
  ssl_certificate_key /etc/letsencrypt/live/concluder.org/privkey.pem; # managed by Certbot
  include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
  ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

  location /.well-known/acme-challenge/ { root /var/www/letsencrypt; }

  # ---- tRPC API -> backend on 4000 ----
  location /api/trpc/ {
    proxy_pass http://127.0.0.1:4000/;
    proxy_http_version 1.1;

    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;

    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection "upgrade";
  }

  # ---- Static SPA ----
  root /root/wiki_concluder/client;
  index index.html;

  location ~* \.(?:css|js|mjs|map|ico|png|jpg|jpeg|gif|svg|webp|woff2?|ttf|eot)$ {
    try_files $uri =404;
    expires 30d;
    add_header Cache-Control "public, max-age=2592000";
  }

  location / {
    try_files $uri $uri/ /index.html;
  }
}

server {
  listen 80;
  listen [::]:80;
  server_name concluder.org;

  if ($host = concluder.org) {
    return 301 https://$host$request_uri;
  } # managed by Certbot

  return 404; # managed by Certbot
}
